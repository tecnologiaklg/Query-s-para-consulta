 JOIN TGFITE ITE ON CAB.NUNOTA = ITE.NUNOTA
 JOIN TGFDIN DIN ON ITE.NUNOTA = DIN.NUNOTA AND ITE.SEQUENCIA = DIN.SEQUENCIA

, CAB.TIPMOV
, PRO.USOPROD

1.65 AS ALIQUOTACALCULADA,
1.65 AS ALIQUOTANF,

ROUND(FIN.VLRDESDOB * 1.65/100,2) AS VALORNF,
'SERVICO DE TRANSPORTE' AS DESCRPROD,
'SERVICO' AS USOPROD,
FIN.CODCFO,
CFO.DESCRCFO
FROM TGFFIN FIN
JOIN TGFTOP TOP ON FIN.CODTIPOPER = TOP.CODTIPOPER AND FIN.DHTIPOPER = TOP.DHALTER
JOIN TSIEMP EMP ON FIN.CODEMP = EMP.CODEMP
JOIN TGFCFO CFO ON FIN.CODCFO = CFO.CODCFO
WHERE FIN.CODTIPOPER IN (4102, 4103)
AND (FIN.DTNEG BETWEEN :PERIODO.INI AND :PERIODO.FIN)
