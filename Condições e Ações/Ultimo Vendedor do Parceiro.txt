SELECT MAX(DECODE (TGFCAB.CODVEND, '1','LEONARDO','2','RUBENS','7','GUSTAVO','13','ADAM','3','MICHEL','12','FERNANDO','14','FELIPE')) 

FROM TGFCAB 

WHERE
TGFCAB.CODTIPOPER IN (3200, 3205, 3250, 3262) 
AND TGFCAB.CODPARC = TGFPAR.CODPARC 
AND TGFCAB.DTNEG = (SELECT MAX(TGFCAB.DTNEG) FROM TGFCAB 
WHERE TGFCAB.CODTIPOPER IN (3200, 3205, 3250, 3262)  AND TGFCAB.CODPARC=TGFPAR.CODPARC)
