--CUS.DTATUAL,
CTE.DTCONTAGEM,
CTE.CODEMP,
CTE.CODLOCAL,
CTE.CODPROD,
CTE.CONTROLE,
PRO.DESCRPROD,
CTE.QTDEST,
CTE.CODVOL,
CTE.TIPO,
--CUS.CUSREP,

TGFCTE CTE
--JOIN TGFCUS CUS ON CTE.CODPROD = CUS.CODPROD AND CTE.CONTROLE = CUS.CONTROLE
