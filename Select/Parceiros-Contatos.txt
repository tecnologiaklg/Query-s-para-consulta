SELECT
PAR.CODPARC,
PAR.RAZAOSOCIAL,
DECODE (PAR.TIPPESSOA,'J','JURIDICA','F','FISICA') AS TIPO_DE_PESSOA,
PAR.CODVEND AS VENDEDOR,
PAR.TELEFONE,
PAR.EMAIL,
VEN.APELIDO,
PAR.CGC_CPF AS CNPJ_CPF,
PAR.AD_UF AS ESTADO,
PAR.IDENTINSCESTAD,
CID.NOMECID AS CIDADE,
CTT.NOMECONTATO,
CTT.EMAIL AS EMAIL_CONTATO,
CTT.TELEFONE AS TEL_CONTATO,
CTT.CELULAR AS CEL_CONTATO



FROM
TGFPAR PAR
LEFT OUTER JOIN TGFCTT CTT ON PAR.CODPARC = CTT.CODPARC
JOIN TGFVEN VEN ON PAR.CODVEND = VEN.CODVEND
JOIN TSICID CID ON PAR.CODCID = CID.CODCID

WHERE
PAR.CLIENTE = 'S' AND PAR.CODPARC <> '0' AND PAR.ATIVO = 'S'
