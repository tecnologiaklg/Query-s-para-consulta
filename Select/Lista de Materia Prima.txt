SELECT
LMP.CODPRODPA,
LMP.CODPRODMP,
PRO2.DESCRPROD,
LMP.QTDMISTURA,
LMP.CODVOL,
LMP.VERIFICAEST,
LMP.GERAREQUISICAO,
PRC.CODPRC,
LPA.IDPROC

FROM
TPRLMP LMP                                              --LISTA DE MAT PRIMA
JOIN TPRLPA LPA ON LMP.CODPRODPA = LPA.CODPRODPA        --LISTA DE PA
JOIN TPRPRC PRC ON LPA.IDPROC = PRC.IDPROC              --PROCESSO PRODUTIVO
JOIN TGFPRO PRO2 ON LMP.CODPRODMP = PRO2.CODPROD

WHERE

PRC.CODPRC = '37' AND LMP.CODPRODPA = '2245'
