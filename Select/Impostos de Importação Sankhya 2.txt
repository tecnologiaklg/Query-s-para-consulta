SELECT
ITE.CODPROD,
ITE.NUNOTA,
ITE.SEQUENCIA,
BASEIMPOSTO,
III.VLRDESPADUA,
III.VLRIMPOSTO,
III.VLRIOF,
IDI.VLRAFRMM,
IDI.VLRCOFINSIMP,
IDI.VLRPISIMP,
ITE.VLRICMS,
ITE.BASEICMS


FROM
TGFITE ITE
LEFT JOIN TGFIII III ON ITE.SEQUENCIA = III.SEQUENCIA
LEFT JOIN TGFIDI IDI ON ITE.SEQUENCIA = IDI.SEQUENCIA

WHERE
III.NUNOTA = '119255'
AND ITE.NUNOTA = '119255'
AND IDI.NUNOTA = '119255'
