SELECT

CAB.NUNOTA,
CAB.NUMNOTA,
CAB.CODPARC,
CAB.CODTIPOPER,
PAR.RAZAOSOCIAL,
CAB.DTNEG,
TOP.DESCROPER,
DECODE (TOP.TIPMOV,'V','VENDA','C','COMPRA','D','DEV VENDA','E','DEV COMPRA')

FROM 
TGFCAB CAB
JOIN TGFPAR PAR ON CAB.CODPARC = PAR.CODPARC
JOIN TGFTOP TOP ON CAB.CODTIPOPER = TOP.CODTIPOPER

WHERE
CAB.NUMNOTA = '1568' AND CAB.CODEMP = '2' AND TOP.CODMODDOC = '55'
