SELECT
ITC.NUMCOTACAO,
ITC.CODPROD,
PRO.DESCRPROD,
ITC.CODPARC,
PAR.RAZAOSOCIAL,
ITC.QTDCOTADA,
ITC.MELHOR,
ITC.SITUACAO,
ITC.VLRDESC,
ITC.CUSTOFINAL,
ITC.PRECO,
ITC.CODMOEDA,
ITC.VLRMOEDA

FROM
TGFITC ITC
JOIN TGFPRO PRO ON ITC.CODPROD = PRO.CODPROD
JOIN TGFPAR PAR ON ITC.CODPARC = PAR.CODPARC

WHERE
ITC.NUMCOTACAO = '83' AND ITC.CODPARC <> '0'
