SELECT
ITE.NUNOTA,
CAB.NUMNOTA,
CAB.DTNEG,
DECODE (CAB.CODTIPOPER,'3200','VENDA COMUM','3205','SIMPLES FAT'),
ITE.CODPROD,
PRO.DESCRPROD,
ITE.VLRTOT,
ITE.QTDNEG

FROM
TGFITE ITE 
JOIN TGFCAB CAB ON ITE.NUNOTA = CAB.NUNOTA 
JOIN TGFPRO PRO ON ITE.CODPROD = PRO.CODPROD

WHERE
PRO.USOPROD IN ('2','V') AND
CAB.DTNEG >= '01/03/2021' AND -- Dt Inicio
CAB.DTNEG <= '29/03/2021' AND -- Dt Fim
CAB.CODTIPOPER IN ('3200','3205')
