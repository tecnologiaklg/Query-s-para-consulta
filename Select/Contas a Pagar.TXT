SELECT DISTINCT
FIN.NUFIN AS NU_FIN,
DECODE (FIN.CODEMP,'1','KLG COMERCIO','2','KLG INDUSTRIA') AS NUM_EMP,
FIN.DTVENC AS DT_VENC,
FIN.NUMNOTA AS NUM_NOTA,
FIN.CODPARC AS COD_PARC,
PAR.RAZAOSOCIAL AS FORNECEDOR,
FIN.VLRDESDOB AS VALOR_PAGAR,
FIN.DHBAIXA AS DATA_BAIXA,
TIT.DESCRTIPTIT AS TIPO_TIT

FROM
TGFFIN FIN
JOIN TGFPAR PAR ON FIN.CODPARC = PAR.CODPARC
JOIN TGFTIT TIT ON FIN.CODTIPTIT = TIT.CODTIPTIT

WHERE
FIN.CODEMP = $P{P_EMP} AND
FIN.DTVENC >= $P{P_DATA_INI} AND --INICIAL
FIN.DTVENC <= $P{P_DATA_FIN} AND --FINAL
FIN.RECDESP = '-1' AND
FIN.PROVISAO = 'N'
